# Define a versão mínima do CMake necessária para o projeto.
cmake_minimum_required(VERSION 3.10)

# Define o nome do projeto e a sua versão.
project(blockchain VERSION 1.0)

# Define o padrão C++ a ser usado.
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Cria um executável com todos os arquivos de código-fonte necessários.
add_executable(blockchain
    src/main.cpp
    src/VirtualMachine.cpp
    src/Token.cpp
)
